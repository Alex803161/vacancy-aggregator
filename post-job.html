// В post-job.html добавить: document.getElementById('jobForm').addEventListener('submit', async (e) => { e.preventDefault(); const formData = new FormData(e.target); const job = { id: Date.now(), title: formData.get('title'), company: formData.get('company'), city: formData.get('city'), salary: formData.get('salary'), description: formData.get('description'), isPremium: formData.get('premium') === 'true', timestamp: Date.now() }; // Сохраняем в localStorage для демонстрации const directJobs = JSON.parse(localStorage.getItem('direct_jobs') || '[]'); directJobs.push(job); localStorage.setItem('direct_jobs', JSON.stringify(directJobs)); alert('Вакансия отправлена на модерацию!'); e.target.reset(); });
